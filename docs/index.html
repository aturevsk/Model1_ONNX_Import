Custom Layer: PatchFlattenLayer

Converts spatial Conv2D output [H,W,C,B] to sequence [C,T,B] with correct row-major ordering (W varies fastest, matching PyTorch).

Key insight: MATLAB's column-major reshape gives H-fastest ordering, but PyTorch expects W-fastest. Fixed by permuting to [C,W,H,B] before reshape.

Custom Layer: AddPositionEmbeddingLayer

Adds a learnable [192×197] embedding to the token sequence. Required because MATLAB's native positionEmbeddingLayer performs lookup (replacement), not addition.

Key insight: Weights [C,T] must be reshaped to [C,1,T] to avoid batch dimension expansion via implicit broadcast.

Debugging Methodology

Three-Way Comparison Framework

Each iteration validated against three references: Python ONNX Runtime (ground truth), MATLAB ONNX import, and native MATLAB rebuild. ONNX Runtime intermediate outputs were extracted by modifying the ONNX graph to expose block-level tensors.

Code Samples

rebuild_vit_native.m — Core Architecture Build 
% Build native ViT-Tiny dlnetwork
layers = [
    imageInputLayer([224 224 3], Normalization="none", Name="input")
    convolution2dLayer(16, 192, Stride=16, Padding=0, Name="patch_conv")
    PatchFlattenLayer(192, "patch_flatten")
    embeddingConcatenationLayer(Name="cls_prepend")
    AddPositionEmbeddingLayer(192, 197, "pos_embed")
];
lgraph = layerGraph(layers);

% Add 12 transformer blocks
for b = 0:11
    prefix = sprintf("block%d_", b);
    % LN1 -> Self-Attention -> Residual Add
    lgraph = addLayers(lgraph, layerNormalizationLayer(Name=prefix+"ln1"));
    lgraph = addLayers(lgraph, selfAttentionLayer(3, 192, Name=prefix+"attn"));
    lgraph = addLayers(lgraph, additionLayer(2, Name=prefix+"add1"));
    % LN2 -> MLP(FC+GELU+FC) -> Residual Add
    lgraph = addLayers(lgraph, layerNormalizationLayer(Name=prefix+"ln2"));
    lgraph = addLayers(lgraph, [
        fullyConnectedLayer(768, Name=prefix+"fc1")
        geluLayer(Name=prefix+"gelu")
        fullyConnectedLayer(192, Name=prefix+"fc2")
    ]);
    lgraph = addLayers(lgraph, additionLayer(2, Name=prefix+"add2"));
    % ... connections omitted for brevity
end

% Final layers
lgraph = addLayers(lgraph, [
    layerNormalizationLayer(Name="final_ln")
    indexing1dLayer("first", Name="cls_extract")
    fullyConnectedLayer(1000, Name="head")
]);
nativeNet = dlnetwork(lgraph);
PatchFlattenLayer.m — Custom Spatial-to-Sequence Layer 
classdef PatchFlattenLayer < nnet.layer.Layer & nnet.layer.Formattable
    properties
        EmbedDim (1,1) double
    end
    methods
        function layer = PatchFlattenLayer(embedDim, name)
            layer.Name = name;
            layer.EmbedDim = embedDim;
        end
        function Z = predict(~, X)
            sz = size(X); batchSize = sz(4);
            X = stripdims(X);
            X = permute(X, [3, 2, 1, 4]);  % [C,W,H,B] - row-major order
            Z = reshape(X, size(X,1), [], batchSize);
            Z = dlarray(Z, "CTB");
        end
    end
end
AddPositionEmbeddingLayer.m — Additive Position Embedding 
classdef AddPositionEmbeddingLayer < nnet.layer.Layer & nnet.layer.Formattable
    properties (Learnable)
        Weights  % [embedDim x seqLen]
    end
    methods
        function layer = AddPositionEmbeddingLayer(embedDim, seqLen, name)
            layer.Name = name;
            layer.Weights = randn(embedDim, seqLen, 'single') * 0.02;
        end
        function Z = predict(layer, X)
            % X: [C, B, T] in CBT format
            posEmb = reshape(layer.Weights, size(layer.Weights,1), 1, []);
            Z = X + posEmb;  % [C, 1, T] broadcasts correctly with [C, B, T]
        end
    end
end
Weight Transfer — QKV Splitting Pattern 
% ONNX stores combined QKV as single MatMul weight [inputDim, 3*embedDim]
qkvW = double(W.(sprintf('onnx__MatMul_%d', qkvIdx(bi))))';  % transpose -> [576, 192]
qW = qkvW(1:192, :);      % Query weights [192, 192]
kW = qkvW(193:384, :);    % Key weights   [192, 192]
vW = qkvW(385:576, :);    % Value weights [192, 192]

% Split biases the same way
qkvB = double(W.(sprintf('blocks_%d_attn_qkv_bias', b)));
qB = qkvB(1:192);  kB = qkvB(193:384);  vB = qkvB(385:576);

% Set on selfAttentionLayer
nativeNet = setWeight(nativeNet, prefix+"attn", "QueryWeights", qW);
nativeNet = setWeight(nativeNet, prefix+"attn", "KeyWeights", kW);
nativeNet = setWeight(nativeNet, prefix+"attn", "ValueWeights", vW);
